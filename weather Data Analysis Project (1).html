import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
data = {
    "Date": [
        "2025-01-01","2025-02-01","2025-03-01",
        "2025-04-01","2025-05-01","2025-06-01",
        "2025-07-01","2025-08-01","2025-09-01",
        "2025-10-01","2025-11-01","2025-12-01"
    ],
    "City": ["Hyderabad"]*12,
    "Temperature": [22,24,28,32,35,30,29,28,27,26,24,22],
    "Humidity": [50,52,55,58,60,70,80,82,78,65,55,50],
    "Rainfall": [5,10,20,35,60,150,220,180,120,50,15,8],
    "Wind Speed": [10,12,15,18,20,22,25,23,20,18,15,12]
}

df = pd.DataFrame(data)
df["Date"] = pd.to_datetime(df["Date"])

df
print(df.head())
print(df.info())
print(df.describe())
avg_temp = df["Temperature"].mean()
print("Average Temperature:", avg_temp)
monthly_rainfall = df[["Date","Rainfall"]]
print(monthly_rainfall)
comparison = df[["Temperature","Humidity"]]
print(comparison)
plt.figure(figsize=(8,5))
plt.plot(df["Date"], df["Temperature"], marker='o')
plt.title("Temperature Trend")
plt.xlabel("Date")
plt.ylabel("Temperature (°C)")
plt.grid(True)
plt.show()
plt.figure(figsize=(8,5))
plt.bar(df["Date"].dt.strftime("%b"), df["Rainfall"])
plt.title("Monthly Rainfall")
plt.xlabel("Month")
plt.ylabel("Rainfall (mm)")
plt.show()
plt.figure(figsize=(8,5))
plt.plot(df["Date"], df["Humidity"], marker='o', label="Humidity")
plt.plot(df["Date"], df["Temperature"], marker='s', label="Temperature")
plt.title("Humidity vs Temperature")
plt.xlabel("Date")
plt.legend()
plt.grid(True)
plt.show()
max_temp = df["Temperature"].max()
min_temp = df["Temperature"].min()

print("Highest Temperature:", max_temp)
print("Lowest Temperature:", min_temp)

print("\nAverage Humidity:", df["Humidity"].mean())
print("Total Rainfall:", df["Rainfall"].sum())
